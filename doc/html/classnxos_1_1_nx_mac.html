Test equality of MAC objects. NOTE: Generic API to check for equality of MAC objects across all languages (esp for langiages that dont support operator overloading)

Returns
true if equality property holds, false otherwise
C++:
NxMacMgr *macMgr = sdk->getMacMgr();
if (macMgr) {
NxMac *mac1 = macMgr->getMac("1.1.1", 2);

if (*mac1 == *mac2)
cout << "Equal" << endl;
}

macMgr = sdk.getMacMgr()
if macMgr:

mac2 = macMgr.getMac("1.1.1", 2)
if mac1 == mac2:
print "Equal"

}
}
